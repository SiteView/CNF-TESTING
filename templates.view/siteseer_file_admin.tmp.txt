function replacechar(text){
rtext=text;
startindex=0;
while(rtext.indexOf("<CRLF>",startindex) != -1){
startindex =rtext.indexOf("<CRLF>",startindex);
rtext = rtext.substring(0,startindex) + "\n" + rtext.substring(startindex+6,rtext.length);

}
return rtext;

}
